React Native Calculator App

This is a beginner React Native calculator project. Users can perform addition, subtraction, multiplication and division operations. The app is built using React Native components like View, Text, TouchableOpacity and useState hook. 

import React, { useState } from "react";
import {
  View,
  Text,
  TouchableOpacity
} from "react-native";

export default function App() {

  const [input, setInput] = useState("");

  const pressHandler = (value) => {
    setInput(input + value);
  };

  const clearHandler = () => {
    setInput("");
  };

  const calculate = () => {
    try {
      setInput(eval(input).toString());
    } catch {
      setInput("Error");
    }
  };

  return (
    <View>
      <Text>{input}</Text>
    </View>
  );
}
